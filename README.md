import matplotlib.pyplot as plt

def generate_plot():
    # Generate a simple plot
    x = [1, 2, 3, 4, 5]
    y = [1, 4, 9, 16, 25]
    plt.plot(x, y)
    plt.xlabel('X-axis')
    plt.ylabel('Y-axis')
    plt.title('Sample Plot')
    plt.savefig('plot.png')

if __name__ == "__main__":
    generate_plot()
